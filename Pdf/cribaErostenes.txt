import java.util.Arrays;

public class CribaEratostenes {
    public static void main(String[] args) {
        int limite = 100; // Queremos los primos hasta el 100
        encontrarPrimos(limite);
    }

    public static void encontrarPrimos(int n) {
        // Creamos un arreglo booleano "esPrimo[0..n]" 
        // Inicialmente asumimos que todos son primos (true)
        boolean[] esPrimo = new boolean[n + 1];
        Arrays.fill(esPrimo, true);

        // El 0 y el 1 no son primos
        esPrimo[0] = false;
        esPrimo[1] = false;

        // Aplicamos la lógica de la criba
        for (int p = 2; p * p <= n; p++) {
            // Si esPrimo[p] no ha cambiado, entonces es un primo
            if (esPrimo[p]) {
                // Actualizamos todos los múltiplos de p empezando desde p*p
                for (int i = p * p; i <= n; i += p) {
                    esPrimo[i] = false;
                }
            }
        }

        // Imprimir los resultados
        System.out.println("Números primos hasta el " + n + ":");
        for (int i = 2; i <= n; i++) {
            if (esPrimo[i]) {
                System.out.print(i + " ");
            }
        }
    }
}
